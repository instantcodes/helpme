First Come First Serve CPU Scheduling

Non-preemptive algorithm — processes execute in arrival order. Simple but can cause the convoy effect where short jobs wait behind long ones.

Key Formulas:
CT = time when process finishes  |  TAT = CT − AT (Turnaround Time)  |  WT = TAT − BT (Waiting Time)

C Code — FCFS Implementation
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    char name[10];
    int at, bt, ct, tat, wt;
} Process;

int cmp(const void*a, const void*b) {
    return ((Process*)a)->at - ((Process*)b)->at;
}

void fcfs(Process *p, int n) {
    qsort(p, n, sizeof(Process), cmp); // sort by arrival time
    int time = 0;
    for (int i=0; i<n; i++) {
        if (time < p[i].at) time = p[i].at; // idle wait
        p[i].ct  = time + p[i].bt;
        p[i].tat = p[i].ct - p[i].at;
        p[i].wt  = p[i].tat - p[i].bt;
        time     = p[i].ct;
    }
}

int main() {
    int n = 4;
    Process p[] = {
        {"P1", 0, 8}, {"P2", 1, 4},
        {"P3", 2, 2}, {"P4", 3, 1}
    };
    fcfs(p, n);
    printf("%-6s %-4s %-4s %-4s %-5s %-5s\n",
           "Proc","AT","BT","CT","TAT","WT");
    int ttat=0, twt=0;
    for(int i=0; i<n; i++){
        printf("%-6s %-4d %-4d %-4d %-5d %-5d\n",
               p[i].name, p[i].at, p[i].bt,
               p[i].ct, p[i].tat, p[i].wt);
        ttat += p[i].tat; twt += p[i].wt;
    }
    printf("\nAvg TAT = %.2f  Avg WT = %.2f\n",
           (float)ttat/n, (float)twt/n);
}
FIFO Page Replacement Memory Management

Replace the page that has been in memory the longest. Simple circular queue approach — when a frame is needed, evict the oldest resident page.

Page Fault occurs when a referenced page is not in any frame.
FIFO uses a queue: first page loaded = first page evicted.
Belady's Anomaly: more frames can sometimes cause more faults in FIFO!
Input Parameters
C Code — FIFO Page Replacement
#include <stdio.h>
#include <string.h>

int isInFrames(int *frames, int n, int page) {
    for(int i=0; i<n; i++)
        if(frames[i] == page) return 1;
    return 0;
}

int fifo(int *refs, int len, int nf) {
    int frames[nf], ptr=0, faults=0;
    memset(frames, -1, sizeof(frames));
    for(int i=0; i<len; i++){
        if(!isInFrames(frames, nf, refs[i])){
            frames[ptr] = refs[i]; // replace oldest
            ptr = (ptr+1) % nf;  // circular pointer
            faults++;
            printf("Page %d → FAULT\n", refs[i]);
        } else {
            printf("Page %d → HIT\n", refs[i]);
        }
    }
    return faults;
}

int main() {
    int refs[] = {0,1,2,0,1,3,0,3,1,2,1};
    int n=11, frames=3;
    int faults = fifo(refs, n, frames);
    printf("\nTotal Page Faults: %d\n", faults);
    printf("Page Hits:   %d\n", n - faults);
}
SSTF Disk Scheduling Disk Management

Shortest Seek Time First — at each step, serve the request closest to the current head position. Reduces average seek time vs FCFS but may cause starvation of distant requests.

Total Head Movement = sum of |head moves| across all requests.
SSTF is greedy — always pick nearest unserved cylinder. May cause starvation for requests far from the center of activity.
Input Parameters
C Code — SSTF Implementation
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

int main() {
    int head = 53;
    int reqs[] = {98,183,41,122,14,124,65,67};
    int n = 8, served[n], total = 0;

    for(int i=0; i<n; i++) served[i] = 0;

    printf("Order: %d", head);
    for(int step=0; step<n; step++){
        int minDist = INT_MAX, idx = -1;
        for(int i=0; i<n; i++){
            if(!served[i]){
                int d = abs(reqs[i] - head);
                if(d < minDist){ minDist=d; idx=i; }
            }
        }
        served[idx] = 1;
        total += minDist;
        head = reqs[idx];
        printf(" → %d", head);
    }
    printf("\nTotal Movement: %d cylinders\n", total);
}
Pipe & Message Queue IPC Inter-Process Communication

IPC allows processes to communicate and synchronize. Simulate how pipe (unidirectional byte stream) and message queue (typed message blocks) work.

Pipe: Unidirectional FIFO byte stream. Parent writes → child reads. Created with pipe(fd).

Message Queue: Messages have a type field. Sender and receiver can be any process with the queue key. Created with msgget(), sent with msgsnd(), received with msgrcv().

Shared Memory: Fastest IPC — processes map same physical memory. Needs semaphore for sync.

C Code — Pipe IPC
#include <stdio.h>
#include <unistd.h>
#include <string.h>

int main() {
    int fd[2];           // fd[0]=read end, fd[1]=write end
    char buf[100];

    if(pipe(fd) == -1) { perror("pipe"); return 1; }

    pid_t pid = fork();
    if(pid == 0) {           // CHILD — reader
        close(fd[1]);        // close write end
        read(fd[0], buf, sizeof(buf));
        printf("Child received: %s\n", buf);
        close(fd[0]);
    } else {                  // PARENT — writer
        close(fd[0]);        // close read end
        char *msg = "Hello from parent!";
        write(fd[1], msg, strlen(msg)+1);
        close(fd[1]);
        wait(NULL);
    }
    return 0;
}
C Code — Message Queue IPC
#include <stdio.h>
#include <sys/ipc.h>
#include <sys/msg.h>
#include <string.h>

struct msgbuf {
    long mtype;
    char mtext[100];
};

int main() {
    key_t key = ftok("/tmp", 65);
    int msgid = msgget(key, 0666 | IPC_CREAT);

    /* SENDER */
    struct msgbuf msg;
    msg.mtype = 1;
    strcpy(msg.mtext, "Hello via message queue!");
    msgsnd(msgid, &msg, sizeof(msg.mtext), 0);
    printf("Sent: %s\n", msg.mtext);

    /* RECEIVER */
    struct msgbuf rcv;
    msgrcv(msgid, &rcv, sizeof(rcv.mtext), 1, 0);
    printf("Received: %s\n", rcv.mtext);

    msgctl(msgid, IPC_RMID, NULL); // destroy queue
    return 0;
}
Banker's Algorithm Deadlock Avoidance

Deadlock avoidance algorithm. Checks if granting a resource request leaves the system in a safe state (a state from which all processes can eventually finish).

Need[i][j] = Max[i][j] − Allocation[i][j]
Safe State: A sequence P₁,P₂,...,Pₙ exists where each process can finish using available + previously released resources.
Safety Algorithm: Try to find a process whose Need ≤ Work; if found, "finish" it, add its allocation to Work. Repeat. If all finish → safe.

C Code — Banker's Safety Algorithm
#include <stdio.h>
#include <string.h>

#define P 5
#define R 3

int alloc[P][R]={{0,1,0},{2,0,0},{3,0,2},{2,1,1},{0,0,2}};
int  max[P][R]={{7,5,3},{3,2,2},{9,0,2},{2,2,2},{4,3,3}};
int  avail[R] = {3,3,2};

int safetyCheck(int seq[]) {
    int work[R], finish[P]={};
    memcpy(work, avail, sizeof(work));
    int count=0;
    while(count < P) {
        int found = 0;
        for(int i=0; i<P; i++) {
            if(finish[i]) continue;
            int j;
            for(j=0; j<R; j++)
                if(max[i][j]-alloc[i][j] > work[j]) break;
            if(j==R) {  // Need ≤ Work
                for(int k=0; k<R; k++) work[k] += alloc[i][k];
                finish[i]=1; seq[count++]=i; found=1;
            }
        }
        if(!found) return 0; // unsafe
    }
    return 1;
}

int main() {
    int seq[P];
    if(safetyCheck(seq)) {
        printf("SAFE. Sequence: ");
        for(int i=0; i<P; i++) printf("P%d ", seq[i]);
    } else printf("UNSAFE STATE — deadlock possible!\n");
}
