FCFS CPU Scheduling
First Come First Serve — processes are executed in arrival order. Non-preemptive. TAT = CT − AT  |  WT = TAT − BT  |  CT[i] = max(CT[i-1], AT[i]) + BT[i]
C #include <stdio.h> int main() { int n = 4; char proc[] = {'P','Q','R','S'}; int at[] = {0, 1, 2, 3}; int bt[] = {5, 3, 8, 2}; int ct[4], tat[4], wt[4]; float avgTAT = 0, avgWT = 0; ct[0] = at[0] + bt[0]; for (int i = 1; i < n; i++) { ct[i] = (ct[i-1] > at[i] ? ct[i-1] : at[i]) + bt[i]; tat[i] = ct[i] - at[i]; wt[i] = tat[i] - bt[i]; avgTAT += tat[i]; avgWT += wt[i]; } tat[0] = ct[0] - at[0]; wt[0] = tat[0] - bt[0]; avgTAT = (avgTAT + tat[0]) / n; avgWT = (avgWT + wt[0]) / n; printf("\nProc AT BT CT TAT WT\n"); for (int i = 0; i < n; i++) printf(" %c %d %d %d %d %d\n", proc[i], at[i], bt[i], ct[i], tat[i], wt[i]); printf("\nAvg TAT = %.2f Avg WT = %.2f\n", avgTAT, avgWT); return 0; }

FIFO Page Replacement
FIFO — replace the page that has been in memory the longest. Uses a queue. A page fault occurs when a required page is not in memory.
C #include <stdio.h> #include <stdbool.h> int fifo(int *refs, int n, int frames) { int mem[frames], ptr = 0, faults = 0; bool inMem; for (int i=0; i<frames; i++) mem[i] = -1; for (int i=0; i<n; i++) { inMem = false; for (int j=0; j<frames; j++) if (mem[j] == refs[i]) { inMem = true; break; } if (!inMem) { // PAGE FAULT mem[ptr] = refs[i]; ptr = (ptr + 1) % frames; faults++; } } return faults; } int main() { int refs[] = {0,1,2,0,1,3,0,3,1,2,1}; int n=11, frames=3; printf("FIFO Page Faults: %d\n", fifo(refs, n, frames)); return 0; }

SSTF Disk Scheduling
SSTF (Shortest Seek Time First) — at each step, service the pending request closest to the current head position. Reduces total head movement vs FCFS.
C #include <stdio.h> #include <stdlib.h> int abs_diff(int a, int b) { return a > b ? a-b : b-a; } int main() { int reqs[] = {98,183,41,122,14,124,65,67}; int n=8, head=53, total=0; bool done[8] = {false}; printf("Head: %d\n", head); for (int step=0; step<n; step++) { int minDist=99999, idx=-1; for (int i=0; i<n; i++) if (!done[i] && abs_diff(head, reqs[i]) < minDist) { minDist = abs_diff(head, reqs[i]); idx = i; } done[idx] = true; total += minDist; printf("Move %d → %d (dist=%d)\n", head, reqs[idx], minDist); head = reqs[idx]; } printf("\nTotal head movement: %d cylinders\n", total); return 0; }

IPC — Pipe & Message Queue
Pipe: Unidirectional byte stream between related processes. Use pipe(fd), fork(), write/read.
Message Queue: Stores typed messages. Use msgget(), msgsnd(), msgrcv().
📌 Pipe — Code
C — Pipe #include <stdio.h> #include <unistd.h> #include <string.h> int main() { int fd[2]; // fd[0]=read, fd[1]=write char buf[50]; pid_t pid; pipe(fd); pid = fork(); if (pid == 0) { // CHILD — reads close(fd[1]); read(fd[0], buf, sizeof(buf)); printf("Child got: %s\n", buf); close(fd[0]); } else { // PARENT — writes close(fd[0]); write(fd[1], "Hello from parent!", 18); close(fd[1]); wait(NULL); } return 0; }

C — msgget / msgsnd / msgrcv #include <sys/ipc.h> #include <sys/msg.h> #include <stdio.h> #include <string.h> struct Message { long mtype; char mtext[50]; }; int main() { key_t key = ftok(".", 'M'); int msgid = msgget(key, 0666 | IPC_CREAT); struct Message msg; // SENDER msg.mtype = 1; strcpy(msg.mtext, "OS Lab is fun!"); msgsnd(msgid, &msg, sizeof(msg.mtext), 0); printf("Sent: %s\n", msg.mtext); // RECEIVER msgrcv(msgid, &msg, sizeof(msg.mtext), 1, 0); printf("Received: %s\n", msg.mtext); msgctl(msgid, IPC_RMID, NULL); // cleanup return 0; }

Banker's Algorithm — Deadlock Avoidance
Banker's Algorithm (Dijkstra): Simulates resource allocation to check if a safe sequence exists before granting any resource.

Key matrices: Allocation, Max, Need = Max − Allocation, Available
C — Banker's Safety Check #include <stdio.h> #include <stdbool.h> #define P 5 #define R 3 bool isSafe(int alloc[P][R], int need[P][R], int avail[R], int safeSeq[P]) { bool finish[P] = {false}; int work[R], count = 0; for (int i=0; i<R; i++) work[i] = avail[i]; while (count < P) { bool found = false; for (int i=0; i<P; i++) { if (finish[i]) continue; int j; for (j=0; j<R; j++) if (need[i][j] > work[j]) break; if (j == R) { // can finish! for (int k=0; k<R; k++) work[k] += alloc[i][k]; safeSeq[count++] = i; finish[i] = true; found = true; } } if (!found) return false; // UNSAFE } return true; }
