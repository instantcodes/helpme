// FCFS — First Come First Serve (Non-Preemptive) #include <stdio.h> int main() { int n = 4; int at[] = {0,1,2,3}; // arrival time int bt[] = {5,3,8,2}; // burst time int ct[4], tat[4], wt[4]; ct[0] = at[0] + bt[0]; for (int i=1; i<n; i++) { ct[i] = (ct[i-1] > at[i] ? ct[i-1] : at[i]) + bt[i]; tat[i] = ct[i] - at[i]; wt[i] = tat[i] - bt[i]; } // Print results... }
